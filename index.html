# Analyse_Arborescence_.py — parcours récursif
def analyser(repertoire):
    resultats = []
    for racine, dossiers, fichiers in os.walk(repertoire):
        for f in fichiers:
            chemin = os.path.join(racine, f)
            taille = os.path.getsize(chemin)
            if taille >= TAILLE_MINI_FICHIER_EN_MEBI_OCTET * 1024 * 1024:
                resultats.append([chemin, taille])
    return resultats[:NB_MAXI_FICHIERS]

# trouver_piste_libre() — logique d'attribution automatique
def trouver_piste_libre(aeroport_arrivee, type_avion, heure_arrivee):
    pistes = Piste.objects.filter(
        aeroport=aeroport_arrivee,
        longueur__gte=type_avion.longueur_piste_necessaire
    )
    for piste in pistes:
        debut = heure_arrivee - timedelta(minutes=10)
        fin   = heure_arrivee + timedelta(minutes=10)
        if not Vol.objects.filter(piste_arrivee=piste,
                datetime_arrivee__gte=debut, datetime_arrivee__lte=fin):
            return piste, heure_arrivee
    return None, heure_arrivee + timedelta(minutes=15)
